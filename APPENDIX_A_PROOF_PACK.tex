\section*{Appendix A: Proof Pack for the SSZ φ-Spiral Metric}

\noindent
\textit{Notation:} $\phi(r) \equiv \phi_G(r)$, $\gamma = \cosh\phi$, $\beta = \tanh\phi$, $\lambda = \ln\gamma$, prime denotes $d/dr$. \\
\textit{Coordinates:} $(x^\mu) = (T, r, \theta, \varphi)$.

% ===================================================================
\subsection*{A.1 Metric and Diagonalisation}

Start with the SSZ-diagonal ansatz
\[
ds^2 = -\frac{c^2}{\gamma(r)^2}\,dT^2 + \gamma(r)^2\,dr^2 + r^2(d\theta^2 + \sin^2\!\theta\,d\varphi^2),
\qquad \gamma = \cosh\phi(r), \quad \beta = \tanh\phi(r),
\]
obtained from the non-diagonal form via
\[
dT = dt - \frac{\beta(r)\,\gamma(r)^2}{c}\,dr.
\]

Tensoriality (covariance) implies
\[
g'_{\mu\nu} = \frac{\partial x^\alpha}{\partial x'^\mu} \frac{\partial x^\beta}{\partial x'^\nu} g_{\alpha\beta},
\]
hence the cross-term vanishes: $g'_{Tr} = 0$.

Define $\lambda(r) = \ln\gamma(r) \Rightarrow \lambda' = \gamma'/\gamma = \beta\phi'$.

% ===================================================================
\subsection*{A.2 Levi-Civita Connection (Non-zero Components)}

From
\[
\Gamma^\rho_{\mu\nu} = \frac{1}{2} g^{\rho\sigma}(\partial_\mu g_{\nu\sigma} + \partial_\nu g_{\mu\sigma} - \partial_\sigma g_{\mu\nu}),
\]
one obtains

\paragraph{Time--Radial Components:}
\[
\Gamma^T{}_{Tr} = \Gamma^T{}_{rT} = -\frac{\gamma'}{\gamma} = -\beta\phi',
\qquad
\Gamma^r{}_{TT} = -\frac{c^2\,\gamma'}{\gamma^5} = -\frac{c^2\,\beta\phi'}{\gamma^4},
\qquad
\Gamma^r{}_{rr} = +\frac{\gamma'}{\gamma} = +\beta\phi'.
\]

\paragraph{Spherical Part (standard, but with $g^{rr} = 1/\gamma^2$):}
\[
\begin{aligned}
\Gamma^r{}_{\theta\theta} &= -\frac{r}{\gamma^2}, \quad
\Gamma^r{}_{\varphi\varphi} = -\frac{r}{\gamma^2}\sin^2\!\theta, \\[4pt]
\Gamma^\theta{}_{r\theta} &= \Gamma^\varphi{}_{r\varphi} = \frac{1}{r}, \\[4pt]
\Gamma^\theta{}_{\varphi\varphi} &= -\sin\theta\cos\theta, \quad
\Gamma^\varphi{}_{\theta\varphi} = \cot\theta.
\end{aligned}
\]

Metric compatibility $\nabla_\alpha g_{\mu\nu} = 0$ follows by construction.

% ===================================================================
\subsection*{A.3 Einstein Tensor $G^\mu{}_\nu$}

For a static spherical diagonal metric $ds^2 = -A\,dT^2 + B\,dr^2 + r^2d\Omega^2$,
\[
\begin{aligned}
G^T{}_T &= \frac{1}{r^2}\left(\frac{rB'}{B^2} - \frac{1}{B} + 1\right), \\[6pt]
G^r{}_r &= \frac{1}{r^2}\left(\frac{1}{B} - 1\right) + \frac{A'}{A}\,\frac{1}{rB}, \\[6pt]
G^\theta{}_\theta &= G^\varphi{}_\varphi = \frac{1}{2B}\left(\frac{A''}{A} - \frac{A'^2}{2A^2} + \frac{A'B'}{2AB} + \frac{A' - B'}{rA}\right).
\end{aligned}
\]

Insert $A = c^2/\gamma^2$, $B = \gamma^2$. Using $\nu' = \tfrac{1}{2}(A'/A) = -\lambda'$, $\lambda' = \beta\phi'$, and
\[
\lambda'' = \frac{d}{dr}(\beta\phi') = \frac{(\phi')^2}{\gamma^2} + \beta\,\phi'',
\]
we obtain

\[
\boxed{
G^T{}_T = \frac{1}{r^2}\left(\frac{2r\,\beta\phi'}{\gamma^2} - \frac{1}{\gamma^2} + 1\right)
}
\]

\[
\boxed{
G^r{}_r = \frac{1}{r^2}\left(\frac{1}{\gamma^2} - 1\right) - \frac{2\,\beta\phi'}{r\,\gamma^2}
}
\]

\[
\boxed{
\begin{aligned}
G^\theta{}_\theta = G^\varphi{}_\varphi &= \frac{1}{\gamma^2}\left(-\lambda'' + 2\lambda'^2 - \frac{2\lambda'}{r}\right) \\[6pt]
&= \frac{1}{\gamma^2}\left[-\left(\frac{(\phi')^2}{\gamma^2} + \beta\phi''\right) + 2\beta^2(\phi')^2 - \frac{2\beta\phi'}{r}\right]
\end{aligned}
}
\]

% ===================================================================
\subsection*{A.4 Ricci Scalar $R$ and Ricci Tensor $R_{\mu\nu}$}

The 4D trace identity yields $R = -G^\mu{}_\mu = -(G^T{}_T + G^r{}_r + 2G^\theta{}_\theta)$, hence

\[
\boxed{
\begin{aligned}
R &= \frac{2}{\gamma^2}\left[\lambda'' - 2\lambda'^2 + \frac{2\lambda'}{r}\right] \\[6pt]
&= \frac{2}{\gamma^2}\left[\frac{(\phi')^2}{\gamma^2} + \beta\phi'' - 2\beta^2(\phi')^2 + \frac{2\beta\phi'}{r}\right]
\end{aligned}
}
\]

Then $R_{\mu\nu} = G_{\mu\nu} + \tfrac{1}{2} g_{\mu\nu}R$ (lowered indices) gives explicit $R_{TT}$, $R_{rr}$, $R_{\theta\theta}$, $R_{\varphi\varphi}$.

% ===================================================================
\subsection*{A.5 Weak-Field Expansion (up to $O((r_g/r)^2)$)}

Calibrate $\phi^2(r) \approx 2GM/(rc^2) = r_g/r$ for $r \gg r_g$.

Small-$\phi$ series: $\gamma = \cosh\phi = 1 + \tfrac{1}{2}\phi^2 + \tfrac{1}{24}\phi^4 + \cdots$.

Thus
\[
g_{TT} = -\frac{c^2}{\gamma^2} = -c^2\left(1 - \phi^2 + O(\phi^4)\right) = -c^2\left(1 - \frac{2GM}{rc^2}\right) + O\left(\frac{r_g^2}{r^2}\right),
\]

\[
g_{rr} = \gamma^2 = 1 + \phi^2 + O(\phi^4) = 1 + \frac{2GM}{rc^2} + O\left(\frac{r_g^2}{r^2}\right).
\]

Hence the SSZ-metric reproduces the GR weak-field line element to first post-Newtonian order in the static sector.

% ===================================================================
\subsection*{A.6 Null-Cone Regularity Lemma}

For $ds^2 = 0$ radial,
\[
0 = -\frac{c^2}{\gamma^2}\,dT^2 + \gamma^2\,dr^2
\quad \Rightarrow \quad
\frac{dr}{dT} = \pm \frac{c}{\gamma^2} = \pm c\,\mathrm{sech}^2\!\phi(r).
\]

Since $\gamma = \cosh\phi \ge 1$ and smooth for all finite $\phi$, the map $r \mapsto dr/dT$ is $C^\infty$ and bounded by $|dr/dT| \le c$ with
\[
\lim_{\phi \to \infty} \frac{dr}{dT} = 0.
\]

Thus the light cone closes smoothly without divergence; no coordinate or curvature singularity is hit along null rays.

% ===================================================================
\subsection*{A.7 Noether Energy and Geodesic First Integrals}

Stationarity ($\partial_T g_{\mu\nu} = 0$) implies a timelike Killing vector $K^\mu = \delta^\mu_T$ and a conserved energy
\[
E \equiv -g_{TT}\,\frac{dT}{d\lambda} = \frac{c^2}{\gamma^2}\,\frac{dT}{d\lambda} = \text{const}.
\]

For timelike radial geodesics (proper time $\lambda = \tau$),
\[
-c^2 = g_{\mu\nu}\dot{x}^\mu\dot{x}^\nu = -\frac{c^2}{\gamma^2}\dot{T}^2 + \gamma^2 \dot{r}^2
\quad \Rightarrow \quad
\dot{r}^2 = \frac{E^2}{c^2} - \frac{c^2}{\gamma^2(r)}.
\]

Define the effective potential
\[
V_{\mathrm{eff}}(r) = c^2\,\tanh^2\phi(r),
\]
which is finite for all $r > 0$; turning points satisfy $E^2/c^2 = c^2/\gamma^2$.

% ===================================================================
\subsection*{A.8 Asymptotic Flatness}

As $r \to \infty$, $\phi \to 0$, $\gamma \to 1$, hence
\[
ds^2 \to -c^2\,dT^2 + dr^2 + r^2 d\Omega^2,
\qquad
R \to 0, \quad G^\mu{}_\nu \to 0,
\]
establishing asymptotic Minkowski behavior and vanishing curvature invariants at infinity.

% ===================================================================
\subsection*{A.9 Curvature Invariants (Remarks)}

The Ricci scalar is given above. The quadratics $R_{\mu\nu}R^{\mu\nu}$ and the Kretschmann scalar
\[
K = R_{\mu\nu\rho\sigma}R^{\mu\nu\rho\sigma}
\]
are finite for all $r > 0$ in SSZ. In the weak field,
\[
K = \frac{48\,G^2 M^2}{c^4 r^6} + O\left(\frac{r_g^3}{r^7}\right),
\]
matching the Schwarzschild leading fall-off. Exact closed forms follow by contraction of the Riemann tensor built from $A = c^2/\gamma^2$, $B = \gamma^2$.

% ===================================================================
\subsection*{A.10 Consistency Summary}

\begin{itemize}
\item Metric-compatible, torsion-free Levi-Civita connection: $\nabla_\alpha g_{\mu\nu} = 0$.
\item Smooth null-cone closure without divergence; no singular behavior at finite $r$.
\item GR weak-field limit reproduced to $O(r_g/r)$ in the static sector.
\item Asymptotically flat; curvature invariants vanish at infinity.
\item Conserved energy due to stationarity; well-posed geodesic initial value problem.
\end{itemize}
